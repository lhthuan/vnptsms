SQL Schema — chạy 1 lần trong Supabase SQL Editor:
-- Chạy toàn bộ script này 1 lần trong Supabase SQL Editor
CREATE TABLE IF NOT EXISTS employees (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL, phone TEXT DEFAULT '',
  role TEXT DEFAULT 'Nhân viên', created_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS pharmacies (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL, code TEXT DEFAULT '',
  phone TEXT DEFAULT '', created_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS shipping_providers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL, code TEXT DEFAULT '',
  created_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS orders (
  id TEXT PRIMARY KEY, name TEXT,
  saved_at TIMESTAMPTZ DEFAULT NOW(), total NUMERIC DEFAULT 0,
  cart JSONB DEFAULT '[]', form JSONB DEFAULT '{}',
  selected_notice INTEGER DEFAULT 0,
  created_by UUID REFERENCES employees(id) ON DELETE SET NULL,
  status TEXT DEFAULT 'quoted',
  erp_code TEXT DEFAULT '',
  pharmacy_id UUID REFERENCES pharmacies(id) ON DELETE SET NULL,
  tracking_code TEXT DEFAULT '',
  shipping_provider_id UUID REFERENCES shipping_providers(id) ON DELETE SET NULL
);
CREATE TABLE IF NOT EXISTS inventory (
  sku TEXT PRIMARY KEY, name TEXT NOT NULL,
  price NUMERIC DEFAULT 0, unit TEXT DEFAULT 'Sp',
  stock INTEGER DEFAULT 0, updated_at TIMESTAMPTZ DEFAULT NOW()
);
-- Thêm cột mới nếu đã tồn tại bảng orders cũ:
ALTER TABLE orders ADD COLUMN IF NOT EXISTS status TEXT DEFAULT 'quoted';
ALTER TABLE orders ADD COLUMN IF NOT EXISTS erp_code TEXT DEFAULT '';
ALTER TABLE orders ADD COLUMN IF NOT EXISTS pharmacy_id UUID REFERENCES pharmacies(id) ON DELETE SET NULL;
ALTER TABLE orders ADD COLUMN IF NOT EXISTS tracking_code TEXT DEFAULT '';
ALTER TABLE orders ADD COLUMN IF NOT EXISTS shipping_provider_id UUID REFERENCES shipping_providers(id) ON DELETE SET NULL;
ALTER TABLE orders ADD COLUMN IF NOT EXISTS status_timestamps JSONB DEFAULT '{}';
-- RLS
ALTER TABLE employees ENABLE ROW LEVEL SECURITY;
ALTER TABLE orders    ENABLE ROW LEVEL SECURITY;
ALTER TABLE inventory ENABLE ROW LEVEL SECURITY;
ALTER TABLE pharmacies ENABLE ROW LEVEL SECURITY;
ALTER TABLE shipping_providers ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_all" ON employees FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY "anon_all" ON orders    FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY "anon_all" ON inventory FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY "anon_all" ON pharmacies FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY "anon_all" ON shipping_providers FOR ALL TO anon USING (true) WITH CHECK (true);
